## 🔄 API Integration Architecture

```
Laravel (Source of Truth)
        ↓
   Scramble
        ↓
   openapi.json
        ↓
   Hey API
        ↓
┌─────────────────────────┐
│  Generated SDK & Types  │
├─────────────────────────┤
│ - TypeScript Types      │
│ - API Service Class     │
│ - React Query Hooks     │
│ - Zod Schemas          │
└─────────────────────────┘
        ↓
   React Components
   (Read-only imports)
```

### Why This Architecture?

1. **Single Source of Truth**: Laravel routes define the API contract
2. **No Manual Typing**: All types generated automatically
3. **Type Safety**: Full TypeScript support end-to-end
4. **Automatic Updates**: Change an endpoint → Types update automatically
5. **Best Practices**: Generated code follows OpenAPI standards
6. **No Duplication**: One spec, many generators possible

## 🛠️ Configuration Files

### heyapi.config.ts

Configures how Hey API generates code from OpenAPI spec:
- Input: `public/openapi.json` (generated by Scramble)
- Output: `resources/js/api/generated/`
- Plugins: TypeScript, SDK, transformers
- Generates: Service class, types, hooks ready

### config/scramble.php

Configures how Scramble generates the OpenAPI spec:
- Output path: `public/openapi.json`
- API info (title, version, description)
- Security schemes (Sanctum tokens)
- Excluded routes (health checks, etc.)

## 🚀 Generating the API SDK

### Step 1: Generate OpenAPI Spec

```bash
# From Laravel routes (uses Scramble)
npm run api:generate-openapi

# Or via PHP command
docker compose exec php php artisan api:generate-openapi
```

Output: `public/openapi.json`

### Step 2: Generate TypeScript SDK

```bash
# Generates types, service class, and React Query hooks
npm run api:generate

# Or just the Hey API generation
docker compose exec node openapi-ts
```

Output: `resources/js/api/generated/*`

### Step 3: Use in Components

```tsx
import { useApiQuery, useApiMutation } from '@/api/hooks/useApi';
import { UserService } from '@/api/generated';

function UsersList() {
  const { data, isLoading } = useApiQuery(
    ['users'],
    () => UserService.getUsers()
  );

  const { mutate: createUser } = useApiMutation(
    (data) => UserService.createUser(data),
    {
      onSuccess: () => {
        // Invalidate cache
        queryClient.invalidateQueries({ queryKey: ['users'] });
      }
    }
  );

  if (isLoading) return <div>Loading...</div>;

  return (
    <div>
      {/* Type-safe rendering - User type from generated types */}
      {data?.map((user) => (
        <div key={user.id}>{user.name}</div>
      ))}
    </div>
  );
}
```

### React Hook Form + Zod

```tsx
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { CreateUserSchema } from '@/schemas';

function CreateUserForm() {
  const { register, handleSubmit, formState: { errors } } = useForm({
    resolver: zodResolver(CreateUserSchema),
  });

  return (
    <form onSubmit={handleSubmit((data) => {
      // data is fully typed as CreateUserRequest
      console.log(data);
    })}>
      <input {...register('name')} />
      {errors.name && <span>{errors.name.message}</span>}
    </form>
  );
}
```

### TanStack Query

```tsx
import { useApiQuery, useApiMutation } from '@/api/hooks/useApi';
import { UserService } from '@/api/generated';

function UserProfile({ userId }: { userId: number }) {
  const { data: user, isLoading, error } = useApiQuery(
    ['user', userId],
    () => UserService.getUser(userId)
  );

  const { mutate: updateUser } = useApiMutation(
    (data) => UserService.updateUser(userId, data),
    {
      onSuccess: () => {
        queryClient.invalidateQueries({ queryKey: ['user', userId] });
      }
    }
  );

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return (
    <div>
      <h1>{user?.name}</h1>
      <button onClick={() => updateUser({ name: 'New Name' })}>
        Update
      </button>
    </div>
  );
}
```

### TanStack Table

```tsx
import { useReactTable, getCoreRowModel } from '@tanstack/react-table';

function UserTable({ users }: { users: User[] }) {
  const table = useReactTable({
    data: users,
    columns: [
      { accessorKey: 'id', header: 'ID' },
      { accessorKey: 'name', header: 'Name' },
      { accessorKey: 'email', header: 'Email' },
    ],
    getCoreRowModel: getCoreRowModel(),
  });

  return (
    <table>
      <thead>
        {table.getHeaderGroups().map(headerGroup => (
          <tr key={headerGroup.id}>
            {headerGroup.headers.map(header => (
              <th key={header.id}>
                {header.isPlaceholder ? null : header.column.columnDef.header}
              </th>
            ))}
          </tr>
        ))}
      </thead>
      <tbody>
        {table.getRowModel().rows.map(row => (
          <tr key={row.id}>
            {row.getVisibleCells().map(cell => (
              <td key={cell.id}>{cell.getValue()}</td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
}
```
